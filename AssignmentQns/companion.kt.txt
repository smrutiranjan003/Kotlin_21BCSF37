class User(val name: String, val age: Int) {
    companion object {
        fun createUser(name: String): User {
            return User(name, 18) 
        }
        fun createUserWithOptionalParameters(name: String = "Jane Doe", age: Int = 40): User {
            return User(name, age)
        }
    }
}

fun main() {
    val user1 = User.createUser("Charlie") // Default method
    println("User 1: ${user1.name}, ${user1.age} years old")
    
    val user2 = User.createUserWithOptionalParameters(name = "Eve")
    println("User 2: ${user2.name}, ${user2.age} years old")
    
    val user3 = User.createUserWithOptionalParameters(age = 35)
    println("User 3: ${user3.name}, ${user3.age} years old")
}
